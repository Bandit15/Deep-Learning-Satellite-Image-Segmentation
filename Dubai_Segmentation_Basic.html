3. Load Images and Masks

Assumes data is in data/Tile X/images/ and data/Tile X/masks/ as in the HuggingFace dataset.

In[8]:
def rgb_mask_to_index(mask_img):
    arr = np.array(mask_img)[:, :, :3]
    mask = np.zeros(arr.shape[:2], dtype=np.uint8)
    for rgb, idx in color2index.items():
        matches = (arr == rgb).all(axis=-1)
        mask[matches] = idx
    return mask

images, masks = [], []
img_size = (256, 256)
data_root = 'data'  # Data folder
tiles = [f for f in os.listdir(data_root) if f.startswith('Tile')]
for tile in sorted(tiles):
    img_dir = os.path.join(data_root, tile, 'images')
    mask_dir = os.path.join(data_root, tile, 'masks')
    for img_file in sorted(os.listdir(img_dir)):
        mask_file = os.path.splitext(img_file)[0] + '.png'
        img_path = os.path.join(img_dir, img_file)
        mask_path = os.path.join(mask_dir, mask_file)
        if not os.path.exists(mask_path):
            continue
        img = Image.open(img_path).convert('RGB').resize(img_size)
        mask = Image.open(mask_path).convert('RGB').resize(img_size)
        images.append(np.array(img) / 255.0)
        masks.append(rgb_mask_to_index(mask))
X = np.stack(images)
y = np.stack(masks)
print('Loaded', X.shape[0], 'images.')

5. Define a Simple UNet

In[10]:
def get_simple_unet(input_shape, num_classes):
    inputs = layers.Input(shape=input_shape)
    c1 = layers.Conv2D(8, 3, activation='relu', padding='same')(inputs)
    p1 = layers.MaxPooling2D(2)(c1)
    c2 = layers.Conv2D(16, 3, activation='relu', padding='same')(p1)
    p2 = layers.MaxPooling2D(2)(c2)
    b = layers.Conv2D(32, 3, activation='relu', padding='same')(p2)
    u1 = layers.UpSampling2D(2)(b)
    u1 = layers.concatenate([u1, c2])
    c3 = layers.Conv2D(16, 3, activation='relu', padding='same')(u1)
    u2 = layers.UpSampling2D(2)(c3)
    u2 = layers.concatenate([u2, c1])
    c4 = layers.Conv2D(8, 3, activation='relu', padding='same')(u2)
    outputs = layers.Conv2D(num_classes, 1, activation='softmax')(c4)
    return models.Model(inputs, outputs)

model = get_simple_unet((256,256,3), num_classes)
model.compile(optimizer='adam', loss='categorical_crossentropy', metrics=['accuracy'])

def decode_mask(mask, index2color):
    if mask.ndim == 3 and mask.shape[-1] == 1:
        mask = mask[..., 0]
    rgb = np.zeros((*mask.shape[:2], 3), dtype=np.uint8)
    for idx, color in index2color.items():
        rgb[mask == idx] = color
    return rgb
